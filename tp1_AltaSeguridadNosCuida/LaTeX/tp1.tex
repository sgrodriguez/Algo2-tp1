\tadAxioma{queTipoHay(p, tripla, u)}{
	\IF( fila?($p$) = alto?($u$) $\lor$ \ col?($p$) = ancho?($u$) ) 
		THEN 
			FueraDeRango
		ELSE
			{\IF( $p$ $\in$ obstaculos?($u$) ) 
				THEN
					Obstaculo
				ELSE
					{\IF($p$ $\in$ damePosicionesAs( $\Pi_{1}$(tripla) ) ) 
						THEN
							Agente
						ELSE
							{\IF( $p$ $\in$ damePosicionesH( $\Pi_{2}$(tripla) ) ) 
								THEN
									Hippie
								ELSE
									{\IF( $p$ $\in$ damePosicionesEst( $\Pi_{3}$(tripla) ) )
										THEN
											Estudiante
										ELSE
											$\emptyset$
									FI}
							FI}
					FI}
			FI}
	FI
}
